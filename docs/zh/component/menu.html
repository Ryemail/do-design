# 使用

<template>
	<div>
		<d-menu
			:default-active="active"
			:router="false"
			:menus="menus"
		></d-menu>
	</div>
</template>

<script>
	export default {
		data() {
			return {
				active: 'link',
				menus: [
					{ path: 'link', title: 'Link' },
					{ path: 'dialog', title: 'Dialog' },
					{ path: 'modal', title: 'Modal' },
					{ path: 'table', title: 'Table' },
					{ path: 'form', title: 'Form' },
					{ path: 'upload', title: 'Upload' },
					{ path: 'tabs', title: 'Tabs' },
				],
			};
		},
	};
</script>

参数 说明 类型 默认值
