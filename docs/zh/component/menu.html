# Menu

# 使用

<template>
	<div>
		<d-menu
			:default-active="active"
			:router="false"
			:menus="menus"
		></d-menu>
	</div>
</template>

<script>
	export default {
		data() {
			return {
				active: 'link',
				menus: [
					{ en_name: 'link', name: 'Link' },
					{ en_name: 'dialog', name: 'Dialog' },
					{ en_name: 'modal', name: 'Modal' },
					{ en_name: 'table', name: 'Table' },
					{ en_name: 'form', name: 'Form' },
					{ en_name: 'upload', name: 'Upload' },
					{ en_name: 'tabs', name: 'Tabs' },
				],
			};
		},
	};
</script>
