# PageTabs 页面 tab 切换

# 基础用法

<template>
	<div>
		<d-page-tabs :default-active.sync="active" :tabs="tabs" />
	</div>
</template>

<script>
	export default {
		data() {
			return {
				tabs: [
					{ title: '首页', to: '/' },
					{ title: 'Link', to: '/link' },
				],
				active: 0,
			};
		},
	};
</script>
