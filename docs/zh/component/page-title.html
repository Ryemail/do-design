	<d-page-title title="测试" :show-back="true"></d-page-title>
</template>

<script>
	export default {
		data() {

# 自定义返回事件

<template>
	<d-page-title title="测试" :show-back="true" :back="onBack"></d-page-title>
</template>

<script>
	export default {
		data() {
			return {
				//
			};
		},
		methods: {
			onBack() {
				alert('点击了返回');
			},
		},
	};
</script>
